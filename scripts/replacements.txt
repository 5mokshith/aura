# git-filter-repo replacements file
# Each line: regex:(PATTERN)==>REPLACEMENT
# Conservative patterns for common leaked secrets in text/markdown/code

# OpenAI
regex:(sk-[A-Za-z0-9]{20,})==>REDACTED

# Google API keys
regex:(AIza[0-9A-Za-z_\-]{35})==>REDACTED

# GitHub Personal Access Token
regex:(ghp_[A-Za-z0-9]{36})==>REDACTED

# Slack tokens
regex:(xox[baprs]-[A-Za-z0-9-]{10,})==>REDACTED

# Private keys (PEM blocks)
regex:(-----BEGIN [A-Z ]*PRIVATE KEY-----[\s\S]*?-----END [A-Z ]*PRIVATE KEY-----)==>REDACTED_PRIVATE_KEY

# Supabase keys in docs/env examples
regex:(NEXT_PUBLIC_SUPABASE_ANON_KEY\s*[:=]\s*['\"]?[A-Za-z0-9._-]+['\"]?)==>NEXT_PUBLIC_SUPABASE_ANON_KEY=REDACTED
regex:(SUPABASE_ANON_KEY\s*[:=]\s*['\"]?[A-Za-z0-9._-]+['\"]?)==>SUPABASE_ANON_KEY=REDACTED
regex:(SUPABASE_SERVICE_ROLE_KEY\s*[:=]\s*['\"]?[A-Za-z0-9._-]+['\"]?)==>SUPABASE_SERVICE_ROLE_KEY=REDACTED

# Google OAuth secrets in docs/env examples
regex:(GOOGLE_CLIENT_SECRET\s*[:=]\s*['\"]?[A-Za-z0-9._-]+['\"]?)==>GOOGLE_CLIENT_SECRET=REDACTED
 
 # Google OAuth values written in prose in docs
 regex:(Client ID\s*:\s*[^\r\n]+)==>Client ID: REDACTED
 regex:(Client Secret\s*:\s*[^\r\n]+)==>Client Secret: REDACTED

 # Google OAuth Client ID exact format anywhere (e.g., in code or docs)
 regex:([0-9]{10,}-[A-Za-z0-9\-]+\.apps\.googleusercontent\.com)==>REDACTED_GOOGLE_CLIENT_ID
 
 # Google OAuth Client Secret common prefix (GOCSPX-...)
 regex:(GOCSPX-[A-Za-z0-9_\-]{10,})==>REDACTED_GOOGLE_CLIENT_SECRET
 
 # Environment variable forms
 regex:(GOOGLE_CLIENT_ID\s*[:=]\s*['\"]?[^'"\r\n]+['\"]?)==>GOOGLE_CLIENT_ID=REDACTED

# Known exact leaked values (from docs)
